S_SellSets:
	.@item_id = getarg(0);
	.@item_cost = getiteminfo(.@item_id, ITEMINFO_BUY);
	.@item_weight = getiteminfo(.@item_id, ITEMINFO_WEIGHT);
	mes "[Madeleine Chu]";
	mes "Quantos"+ (.@item_id == 12125 ? "Ar livre":"Interior");
	mes "Kits de culinária seriam";
	mes "você gosta de comprar? ";
	mes "deseja cancelar, por favor";
	mes "digite o número 0.";
	next;
	while(1) {
		input .@sell;
		if (.@sell == 0) {
			mes "[Madeleine Chu]";
			mes "Você mudou seu";
			mes "mente? ";
			mes "comprar"+ getitemname(.@item_id);
			mes "Kits depois é só voltar";
			mes "para mim a qualquer momento, certo?";
			close;
		}
		else if (.@sell > 100) {
			mes "[Madeleine Chu]";
			mes "Ah, me desculpe, mas";
			mes "Eu não vendo mais do que";
			mes "100"+ getitemname(.@item_id);
			mes "de cada vez, apenas por segurança.";
			next;
		}
		else break;
	}
	set .@total_cost,.@sell * .@item_cost;
	set .@total_weight,.@sell * .@item_weight;
	if (Zeny < .@total_cost) {
		mes "[Madeleine Chu]";
		mes "Ah, me desculpe, mas você";
		mes "não posso pagar tantos";
		mes getitemname(.@item_id);
		mes "Por favor, verifique seu zeny";
		mes "antes de comprar meus kits~";
		close;
	}
	if (!checkweight(.@item_id,.@sell)) {
		mes "[Madeleine Chu]";
		mes "Me desculpe, mas você não";
		mes "tenha espaço suficiente em seu";
		mes "Inventário para tantos";
		mes getitemname(.@item_id) +"...";
		close;
	}
	set Zeny, Zeny-.@total_cost;
	getitem .@item_id,.@sell;
	mes "[Madeleine Chu]";
	mes "Aqui está você ~";
	mes "Boa sorte com";
	mes "sua formação culinária!";
	return;
}
